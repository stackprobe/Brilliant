		C:\Factory\SubTools\CsProjAdjustCs.exe Game0001 Common
		ROBOCOPY C:\Dev\Fairy\Donut2\Resource\General ..\Resource\General /MIR

	新
		C:\Factory\SubTools\CopyLib.exe C:\Dev\Fairy\Donut2\Donut2\Donut2\Common Game0001\Common
		C:\Factory\SubTools\CopyLib.exe C:\Dev\Game3\Brilliant\Brilliant\Brilliant\Game3Common Game0001\Game3Common
		C:\Factory\SubTools\CsProjAdjustCs.exe Game0001 Common
		C:\Factory\SubTools\CsProjAdjustCs.exe Game0001 Game3Common
